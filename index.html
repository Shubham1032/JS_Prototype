        let parent={
            name:"shubham",
            age:24
        }
        let child={
            name:"girl"
        }
        child.__proto__=parent;
        console.log(child.age);

Q2. Write code to explain prototype chaining

        let parent={

            age:24,
            city:jaipur
        }
        let child1={

        }
        let child2={
            name:"Monu",
            age:24
        }
        let child3={
            name:"sonu"
        }
        child1.__proto__=parent;
        child2.__proto__=child1;
        child3.__proto__=child2;
        console.log(child3.city);
